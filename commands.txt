# Build Docker Image:
docker build -t <docker_image_name> .

#Run the Container:
docker run -p 5000:5000 <docker_image_name>

# Image Tagging:
docker tag <docker_image_name> <profile_name>/<docker_image_name>:<tag_name>

# Push the Image:
docker push <profile_name>/<docker_image_name>:<tag_name>

# Pull the Image:
docker pull <profile_name>/<docker_image_name>:<tag_name>
